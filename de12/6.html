1. import os
  2. from PIL import Image
  4. # ファイル名を宣言
  5. FromImgName = 'img'
  6. ToImgName = 'resize'
  8. # imgフォルダ内の画像名をまとめて取得
  9. files = os.listdir(FromImgName)
  12. def scale_to_width(img, width): # アスペクト比を固定して、幅が指定した値になるようリサイズする。
  13.     height = round(img.height * width / img.width)
  14.     return img.resize((width, height))
  17. # for文で画像サイズを一括変更
  18. for file in files:
  19.     img = Image.open(os.path.join(FromImgName, file)) # 画像のパスを生成し、imgへ画像を格納
  20.     file_name = os.path.basename(file)
  21.     img_resize = scale_to_width(img, 250)
  22.     img_resize.save('resize/img_resized_{}'.format(file_name), quality=95)
